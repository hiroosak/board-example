/**

 * @api public

angular.module('boardApp.controllers', []).
  controller('HomeCtrl', ['$scope',
    function($scope) {
    }
  ]).
  controller('BoardShowCtrl', ['$scope', '$http', '$routeParams', 'Board',
    function($scope, $http, $routeParams, Board) {
      Board.get({boardId: $routeParams.boardId}, function(data) {
        $scope.board   = data.board;
        $scope.threads = data.threads;
      });
    }
  ]).
  controller('ThreadCreateCtrl', ['$scope', '$http', '$routeParams',
    function($scope, $http, $routeParams) {
      $scope.boardId = $routeParams.boardId;
    }
  ]).
  controller('ThreadShowCtrl', ['$scope', '$http', '$routeParams', 'Thread',
    function($scope, $http, $routeParams, Thread) {
      Thread.get({threadId: $routeParams.threadId}, function(data) {
        $scope.thread    = data.thread;
        $scope.responses = data.responses;
      });
    }
  ]);

angular.module('boardApp.partials', []).
  controller('HeaderNavRightCtrl', ['$scope', 'User', 
    function($scope, User) {
      User.me(function(data) { // success
        $scope.isLogin = true;
      }, function(data) {
        $scope.isLogin = false; // error
      });
    }
  ]).
  controller('SideMenuCtrl', ['$scope', 'Category',

        $scope.categories = data;
      });
    }
